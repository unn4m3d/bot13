        @version = v
        @author = a
        if l.class == "Integer"
                @locale = l
        else
                if l == "RUS"
                        @locale = LOCALE::RUS
                else
                        @locale = LOCALE::ENG
                end 
        end

def older?(v)
        for i in (0...v.length)
                if i < @version.length
                        if v[i].chr.to_i > @version[i].chr.to_i
                                return true
                        elsif v[i].chr.to_i < @version[i].chr.to_i
                                return false
                        end
                end
        end
        return false
end
