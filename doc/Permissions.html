def self.load
        f = File.open($home + "/.bot13/perms.cfg")
        while not f.eof?
                s = f.gets.split(" ")
                $perms[s[0]] = s[1].to_i
        end
        f.close
end
    save() click to toggle source

    Saves permissions to a config

    # File api.rb, line 156
def self.save()
        if not $perms[".default"]
                $perms[".default"] = 0
        end
        f = File.open($home + "/.bot13/perms.cfg","w")
        for k in $perms.keys()
                f.write(k + " " + $perms[k].to_s + "\n") 
        end
        f.close
end
