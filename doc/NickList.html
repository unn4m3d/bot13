def initialize(s,c=nil)
        unless c
                @chan = c
                @serv = s
        else
                @chan = s
                @serv = nil
        end
        return unless $conn
        for k in $conn.channels
                if k.name == @chan then
                        @cdata = k
                end
        end
        @cdata = nil
end

def list
        IRCConnection.send_to_server("NAMES #{@chan}")
        while not $udata
        end
        l,$udata =  $udata.split(" "),nil
        return l
end

def search(nick)
        for u in list
                if u == nick or u == nick.gsub(/[@%+]/,'') then
                        return true
                end
        end
        return false
end
